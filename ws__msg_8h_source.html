177  /// <summary>
178  /// Extracts the body of the incoming message as a string value, only if the message type is UTF-8.
179  /// A body can only be extracted once because in some cases an optimization is made where the data is 'moved' out.
180  /// </summary>
181  /// <returns>String containing body of the message.</returns>
182  _ASYNCRTIMP pplx::task<std::string> extract_string() const;
183 
184  /// <summary>
185  /// Produces a stream which the caller may use to retrieve body from an incoming message.
186  /// Can be used for both UTF-8 (text) and binary message types.
187  /// </summary>
188  /// <returns>A readable, open asynchronous stream.</returns>
189  /// <remarks>
190  /// This cannot be used in conjunction with any other means of getting the body of the message.
191  /// </remarks>
192  concurrency::streams::istream body() const
193  {
194  auto to_uint8_t_stream = [](const concurrency::streams::streambuf<uint8_t> &buf) -> concurrency::streams::istream
195  {
196  return buf.create_istream();
197  };
198  return to_uint8_t_stream(m_body);
199  }
200 
201  /// <summary>
202  /// Returns the length of the received message.
203  /// </summary>
204  size_t length() const
205  {
206  return static_cast<size_t>(m_body.size());
207  }
208
