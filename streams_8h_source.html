204  /// <summary>
205  /// Write a number of characters from a given stream buffer into the stream.
206  /// </summary>
207  /// <param name="source">A source stream buffer.</param>
208  /// <param name="count">The number of characters to write.</param>
209  pplx::task<size_t> write(streams::streambuf<CharType> source, size_t count) const
210  {
211  pplx::task<size_t> result;
212  if ( !_verify_and_return_task(details::_out_stream_msg, result) ) return result;
213  if ( !source.can_read() )
214  return pplx::task_from_exception<size_t>(std::make_exception_ptr(std::runtime_error("source buffer not set up for input of data")));
215 
216  if (count == 0)
217  return pplx::task_from_result((size_t)0);
218 
219  auto buffer = helper()->m_buffer;
220  auto data = buffer.alloc(count);
221 
222  if ( data != nullptr )
223  {
224  auto post_read =
225  [buffer](pplx::task<size_t> op)-> pplx::task<size_t>
226  {
227  auto b = buffer;
228  b.commit(op.get());
229  return op;
230  };
231  return source.getn(data, count).then(post_read);
232  }
233  else
234  {
235  size_t available = 0;
236 
237  const bool acquired = source.acquire(data, available);
238  if (available >= count)
239  {
240  auto post_write =
241  [source,data](pplx::task<size_t> op)-> pplx::task<size_t>
242  {
243  auto s = source;
244  s.release(data,op.get());
245  return op;
246  };
247  return buffer.putn(data, count).then(post_write);
248  }
249  else
250  {
251  // Always have to release if acquire returned true.
252  if(acquired)
253  {
254  source.release(data, 0);
255  }
256 
257  std::shared_ptr<CharType> buf(new CharType[count], [](CharType *buf) { delete [] buf; });
258 
259  auto post_write =
260  [buf](pplx::task<size_t> op)-> pplx::task<size_t>
261  {
262  return op;
263  };
264  auto post_read =
265  [buf,post_write,buffer](pplx::task<size_t> op) -> pplx::task<size_t>
266  {
267  auto b = buffer;
268  return b.putn(buf.get(), op.get()).then(post_write);
269  };
270 
271  return source.getn(buf.get(), count).then(post_read);
272  }
273  }
