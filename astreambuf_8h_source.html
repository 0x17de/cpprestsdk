306  //
307  // Efficient read and write.
308  //
309  // The following routines are intended to be used for more efficient, copy-free, reading and
310  // writing of data from/to the stream. Rather than having the caller provide a buffer into which
311  // data is written or from which it is read, the stream buffer provides a pointer directly to the
312  // internal data blocks that it is using. Since not all stream buffers use internal data structures
313  // to copy data, the functions may not be supported by all. An application that wishes to use this
314  // functionality should therefore first try them and check for failure to support. If there is
315  // such failure, the application should fall back on the copying interfaces (putn / getn)
316  //

396  /// <summary>
397  /// Closes the stream buffer, preventing further read or write operations.
398  /// </summary>
399  /// <param name="mode">The I/O mode (in or out) to close for.</param>
400  virtual pplx::task<void> close(std::ios_base::openmode mode = std::ios_base::in | std::ios_base::out)
401  {
402  pplx::task<void> closeOp = pplx::task_from_result();
403 
404  if (mode & std::ios_base::in && can_read()) {
405  closeOp = _close_read();
406  }
407 
408  // After the flush_internal task completed, "this" object may have been destroyed,
409  // accessing the members is invalid, use shared_from_this to avoid access violation exception.
410  auto this_ptr = std::static_pointer_cast<streambuf_state_manager>(this->shared_from_this());
411 
412  if (mode & std::ios_base::out && can_write()) {
413  if (closeOp.is_done())
414  closeOp = closeOp && _close_write().then([this_ptr]{}); // passing down exceptions from closeOp
415  else
416  closeOp = closeOp.then([this_ptr] { return this_ptr->_close_write().then([this_ptr]{}); });
417  }
418 
419  return closeOp;
420  }

723  // Set exception and eof states for async read
724  template<typename _CharType1>
725  pplx::task<_CharType1> create_exception_checked_task(pplx::task<_CharType1> result, std::function<bool(_CharType1)> eof_test, std::ios_base::openmode mode = std::ios_base::in | std::ios_base::out)
726  {
727  auto thisPointer = this->shared_from_this();
728 
729  auto func1 = [=](pplx::task<_CharType1> t1) -> pplx::task<_CharType1> {
730  try {
731  thisPointer->m_stream_read_eof = eof_test(t1.get());
732  } catch (...) {
733  thisPointer->close(mode, std::current_exception()).get();
734  return pplx::task_from_exception<_CharType1>(thisPointer->exception(), pplx::task_options());
735  }
736  if (thisPointer->m_stream_read_eof && !(thisPointer->exception() == nullptr))
737  return pplx::task_from_exception<_CharType1>(thisPointer->exception(), pplx::task_options());
738  return t1;
739  };
740 
741  if ( result.is_done() )
742  {
743  // If the data is already available, we should avoid scheduling a continuation, so we do it inline.
744  return func1(result);
745  }
746  else
747  {
748  return result.then(func1);
749  }
750  }
