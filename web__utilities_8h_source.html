167 /// <summary>
168 /// web_proxy represents the concept of the web proxy, which can be auto-discovered,
169 /// disabled, or specified explicitly by the user.

171 class web_proxy
172 {
173  enum web_proxy_mode_internal{ use_default_, use_auto_discovery_, disabled_, user_provided_ };
174 public:
175  enum web_proxy_mode{ use_default = use_default_, use_auto_discovery = use_auto_discovery_, disabled = disabled_};
176 
177  /// <summary>
178  /// Constructs a proxy with the default settings.
179  /// </summary>
180  web_proxy() : m_address(_XPLATSTR("")), m_mode(use_default_) {}
181 
182  /// <summary>
183  /// Creates a proxy with specified mode.
184  /// </summary>
185  /// <param name="mode">Mode to use.</param>
186  web_proxy( web_proxy_mode mode ) : m_address(_XPLATSTR("")), m_mode(static_cast<web_proxy_mode_internal>(mode)) {}
187 
188  /// <summary>
189  /// Creates a proxy explicitly with provided address.
190  /// </summary>

192  web_proxy( uri address ) : m_address(address), m_mode(user_provided_) {}
193 
194  /// <summary>
195  /// Gets this proxy's URI address. Returns an empty URI if not explicitly set by user.
196  /// </summary>
197  /// <returns>A reference to this proxy's URI.</returns>
198  const uri& address() const { return m_address; }
199 
200  /// <summary>
201  /// Gets the credentials used for authentication with this proxy.
202  /// </summary>
203  /// <returns>Credentials to for this proxy.</returns>
204  const web::credentials& credentials() const { return m_credentials; }
205 
206  /// <summary>
207  /// Sets the credentials to use for authentication with this proxy.
208  /// </summary>
209  /// <param name="cred">Credentials to use for this proxy.</param>
210  void set_credentials(web::credentials cred) {
211  if( m_mode == disabled_ )
212  {
213  throw std::invalid_argument("Cannot attach credentials to a disabled proxy");
214  }
215  m_credentials = std::move(cred);
216  }
217 
218  /// <summary>
219  /// Checks if this proxy was constructed with default settings.
220  /// </summary>
221  /// <returns>True if default, false otherwise.</param>
222  bool is_default() const { return m_mode == use_default_; }
223 
224  /// <summary>
225  /// Checks if using a proxy is disabled.
226  /// </summary>
227  /// <returns>True if disabled, false otherwise.</returns>
228  bool is_disabled() const { return m_mode == disabled_; }
229 
230  /// <summary>
231  /// Checks if the auto discovery protocol, WPAD, is to be used.
232  /// </summary>
233  /// <returns>True if auto discovery enabled, false otherwise.</returns>
234  bool is_auto_discovery() const { return m_mode == use_auto_discovery_; }
235 
236  /// <summary>
237  /// Checks if a proxy address is explicitly specified by the user.
238  /// </summary>
239  /// <returns>True if a proxy address was explicitly specified, false otherwise.</returns>
240  bool is_specified() const { return m_mode == user_provided_; }
241 
242 private:
243  web::uri m_address;
244  web_proxy_mode_internal m_mode;
245  web::credentials m_credentials;
246 };
