283  /// <summary>
284  /// Gets a number of characters from the buffer and place it into the provided memory block.
285  /// </summary>
286  std::streamsize xsgetn(_Out_writes_ (count) CharType* ptr, _In_ std::streamsize count)
287  {
288  size_t cnt = size_t(count);
289  size_t read_so_far = 0;
290 
291  try
292  {
293  while (read_so_far < cnt)
294  {
295  size_t rd = m_buffer.getn(ptr+read_so_far, cnt-read_so_far).get();
296  read_so_far += rd;
297  if ( rd == 0 )
298  break;
299  }
300  return read_so_far;
301  }
302  catch(...)
303  {
304  return 0;
305  }
306  }
