345  static void verify_uri(const web::uri& uri)
346  {
347  // Most of the URI schema validation is taken care by URI class.
348  // We only need to check certain things specific to websockets.
349  if(uri.scheme() != _XPLATSTR("ws") && uri.scheme() != _XPLATSTR("wss"))
350  {
351  throw std::invalid_argument("URI scheme must be 'ws' or 'wss'");
352  }
353 
354  if(uri.host().empty())
355  {
356  throw std::invalid_argument("URI must contain a hostname.");
357  }
358 
359  // Fragment identifiers are meaningless in the context of WebSocket URIs
360  // and MUST NOT be used on these URIs.
361  if (!uri.fragment().empty())
362  {
363  throw std::invalid_argument("WebSocket URI must not contain fragment identifiers");
364  }
365  }
366 
367 protected:
368  web::uri m_uri;
369  websocket_client_config m_config;
370 };
