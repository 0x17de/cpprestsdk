505 // utf16char is not fully supported in GCC
506 class cmp
507 {
508 public:
509 
510  static int icmp(std::string left, std::string right)
511  {
512  size_t i;
513  for (i = 0; i < left.size(); ++i)
514  {
515  if (i == right.size()) return 1;
516 
517  auto l = cmp::tolower(left[i]);
518  auto r = cmp::tolower(right[i]);
519  if (l > r) return 1;
520  if (l < r) return -1;
521  }
522  if (i < right.size()) return -1;
523  return 0;
524  }
525 
526 private:
527  static char tolower(char c)
528  {
529  if (c >= 'A' && c <= 'Z')
530  return static_cast<char>(c - 'A' + 'a');
531  return c;
532  }
533 };
