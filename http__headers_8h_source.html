217  /// <summary>
218  /// Attempts to match a header field with the given name using the '>>' operator.
219  /// </summary>
220  /// <param name="name">The name of the header field.</param>
221  /// <param name="value">The value of the header field.</param>
222  /// <returns><c>true</c> if header field was found and successfully stored in value parameter.</returns>
223  template<typename _t1>
224  bool match(const key_type &name, _t1 &value) const
225  {
226  auto iter = m_headers.find(name);
227  if (iter != m_headers.end())
228  {
229  // Check to see if doesn't have a value.
230  if(iter->second.empty())
231  {
232  bind_impl(iter->second, value);
233  return true;
234  }
235  return bind_impl(iter->second, value);
236  }
237  else
238  {
239  return false;
240  }
241  }
