70 // Helper function to trim leading and trailing null characters from a string.
71 static void trim_nulls(utility::string_t &str)
72 {
73  size_t index;
74  for(index = 0; index < str.size() && str[index] == 0; ++index);
75  str.erase(0, index);
76  for(index = str.size(); index > 0 && str[index - 1] == 0; --index);
77  str.erase(index);

80 // Flatten the http_headers into a name:value pairs separated by a carriage return and line feed.
81 static utility::string_t flatten_http_headers(const http_headers &headers)

83  utility::string_t flattened_headers;

98 static void parse_headers_string(_Inout_z_ utf16char *headersStr, http_headers &headers)
99 {
100  utf16char *context = nullptr;
101  utf16char *line = wcstok_s(headersStr, CRLF, &context);
102  while(line != nullptr)
103  {
104  const utility::string_t header_line(line);
105  const size_t colonIndex = header_line.find_first_of(_XPLATSTR(":"));
106  if(colonIndex != utility::string_t::npos)
107  {
108  utility::string_t key = header_line.substr(0, colonIndex);
109  utility::string_t value = header_line.substr(colonIndex + 1, header_line.length() - colonIndex - 1);
110  http::details::trim_whitespace(key);
111  http::details::trim_whitespace(value);
112  headers.add(key, value);
113  }
114  line = wcstok_s(nullptr, CRLF, &context);
115  }
116 }

324  // Wraps opening the client around sending a request.
325  void open_and_send_request(std::shared_ptr<request_context> request)
326  {
327  // First see if client needs to be opened.
328  auto error = open_if_required();
329 
330  if (error != 0)
331  {
332  // Failed to open
333  request->report_error(error, _XPLATSTR("Open failed"));
334 
335  // DO NOT TOUCH the this pointer after completing the request
336  // This object could be freed along with the request as it could
337  // be the last reference to this object
338  return;
339  }
340 
341  send_request(request);
342  }

344  unsigned long open_if_required()
345  {
346  unsigned long error = 0;
347 
348  if(!m_opened)
349  {
350  pplx::extensibility::scoped_critical_section_t l(m_open_lock);
351 
352  // Check again with the lock held
353  if (!m_opened)

367  void push_request(_In_opt_ std::shared_ptr<request_context> request)
368  {
369  if (request == nullptr) return;
370 
371  pplx::extensibility::scoped_critical_section_t l(m_open_lock);
372 
373  if(++m_scheduled == 1)
374  {
375  // Schedule a task to start sending.
376  pplx::create_task([this, request]() -> void
377  {
378  open_and_send_request(request);
379  });
380  }
381  else
382  {
383  m_requests_queue.push(request);
384  }
385  }

388  std::queue<std::shared_ptr<request_context>> m_requests_queue;
